/*CREATING A TABLE FOR INSERTING THE DETAILS OF THE TRAIN*/

CREATE TABLE TRAIN
(
    TRAIN_NUM INT,
    TRAIN_NAME VARCHAR(500),
    TRAIN_IN_TIME DATETIME,
    TRAIN_OUT_TIME DATETIME,
    STATION_CODE INT,
    STATION_NAME VARCHAR(500),
    DESTINATION VARCHAR(500)
)

/* Adding the Primary key constraint to the Table named Train */
ALTER TABLE TRAIN
ALTER COLUMN DESTINATION VARCHAR(500) NOT NULL;

ALTER TABLE TRAIN
ADD CONSTRAINT PK_TRAIN_DESTINATION PRIMARY KEY (DESTINATION);
